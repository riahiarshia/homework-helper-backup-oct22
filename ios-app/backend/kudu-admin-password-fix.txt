// Simple Admin Password Change Script for Kudu Console
// Copy everything below this line into a file called "fix-password.js" in Kudu console

const crypto = require('crypto');
const { Pool } = require('pg');

async function changeAdminPassword() {
  const pool = new Pool({
    connectionString: process.env.DATABASE_URL,
    ssl: process.env.NODE_ENV === 'production' || process.env.NODE_ENV === 'staging' ? { rejectUnauthorized: false } : false
  });

  try {
    // New password
    const newPassword = 'Admin123!Staging';
    const passwordHash = crypto.createHash('sha256').update(newPassword).digest('hex');
    
    console.log('Changing admin password to:', newPassword);
    
    // Update the password
    const result = await pool.query(
      'UPDATE admin_users SET password_hash = $1 WHERE username = $2 RETURNING username, email',
      [passwordHash, 'admin']
    );
    
    if (result.rows.length > 0) {
      console.log('✅ Password updated successfully!');
      console.log('Username:', result.rows[0].username);
      console.log('Email:', result.rows[0].email);
      console.log('New Password:', newPassword);
    } else {
      console.log('❌ No admin user found with username "admin"');
    }
    
  } catch (error) {
    console.error('❌ Error:', error.message);
  } finally {
    await pool.end();
  }
}

changeAdminPassword();
